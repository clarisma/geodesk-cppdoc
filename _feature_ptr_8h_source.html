27class GEODESK_API FeaturePtr
28{
29public:
30 FeaturePtr() {}
31 FeaturePtr(const uint8_t* p) : p_(p) {}
32 FeaturePtr(const FeaturePtr& other) : p_(other.p_) {}
33
34 operator DataPtr () const noexcept { return p_; }
35
36 uint64_t id() const noexcept
37 {
38 return p_.getUnsignedLongUnaligned() >> 12;
39 }
40
41 TypedFeatureId typedId() const noexcept
42 {
43 return TypedFeatureId::ofTypeAndId(type(), id());
44 }
45
46 FeatureHeader header() const noexcept
47 {
48 return FeatureHeader(p_.getUnsignedLongUnaligned());
49 }
50
51 // TODO: should just get pointer
52 Box bounds() const noexcept
53 {
54 assert(!isNode());
55 return Box((p_-16).getInt(), (p_-12).getInt(), (p_-8).getInt(), (p_-4).getInt());
56 }
57
58 Coordinate bottomLeft() const noexcept
59 {
60 return Coordinate(
61 (p_ - (isNode() ? 8 : 16)).getInt(),
62 (p_ - (isNode() ? 4 : 12)).getInt());
63 }
64
65 Coordinate topRight() const noexcept
66 {
67 return Coordinate(
68 (p_ - 8).getInt(),
69 (p_ - 4).getInt());
70 }
71
72 bool isNull() const noexcept { return !p_; }
73 int flags() const noexcept { return p_.getInt(); }
74 bool isArea() const noexcept { return p_.getUnsignedInt() & FeatureFlags::AREA; }
75 bool isNode() const noexcept { return type() == FeatureType::NODE; }
76 bool isWay() const noexcept { return type() == FeatureType::WAY; }
77 bool isRelation() const noexcept { return type() == FeatureType::RELATION; }
78 bool isRelationMember() const noexcept { return flags() & FeatureFlags::RELATION_MEMBER; }
79 bool isType(FeatureTypes types) const noexcept { return types.acceptFlags(flags()); }
80
81 // Placeholder features no longer exist in v2
82 // But we may need a similar mechanism once we introduce deleted features,
83 // so keep this for now but always return false
84 bool isPlaceholder() const { return false; }
85
86 // TODO: check for type==node as well if we ever re-purpose
87 // the exception_node flag for other feature types (can be done
88 // with a simple mask op)
89 bool isExceptionNode() const
90 {
91 return flags() & FeatureFlags::EXCEPTION_NODE;
92 }
93
94 // TODO: In v2, these bits are re-used for node flags
95 // colocated_node / exception_node
96 // No: these flags have been moved to bit 8 / bit 9
97 bool hasNorthwestTwin() const noexcept
98 {
99 return flags() & (FeatureFlags::MULTITILE_NORTH | FeatureFlags::MULTITILE_WEST);
100 }
101
102 bool intersects(const Box& bounds) const noexcept
103 {
104 assert(!isNode());
105 return (!((p_-16).getInt() > bounds.maxX() ||
106 (p_-12).getInt() > bounds.maxY() ||
107 (p_-8).getInt() < bounds.minX() ||
108 (p_-4).getInt() < bounds.minY()));
109 }
110
111
112
113 TagTablePtr tags() const noexcept
114 {
115 return TagTablePtr::readFrom(p_ + 8);
116 }
117
118 DataPtr ptr() const noexcept { return p_; }
119 DataPtr bodyptr() const noexcept { return (p_ + 12).follow(); }
120
126 DataPtr relationTableFast() const
127 {
128 assert(isRelationMember());
129 // For nodes, the body pointer points to the reltable
130 // For ways and relations, the reltable pointer sits right
131 // before the body anchor
132 DataPtr ppRelTable = isNode() ? (p_ + 12) : (bodyptr() - 4);
133 return ppRelTable.followUnaligned();
134 // TODO: unaligned only for ways/relations
135 }
136
137 int32_t relationTableHandleFast(const uint8_t* base) const
138 {
139 assert(isRelationMember());
140 // For nodes, the body pointer points to the reltable
141 // For ways and relations, the reltable pointer sits right
142 // before the body anchor
143 DataPtr ppRelTable = isNode() ? (p_ + 12) : (bodyptr() - 4);
144 return static_cast<int32_t>((ppRelTable - base) +
145 ppRelTable.getIntUnaligned());
146 }
147
154 int64_t idBits() const
155 {
156 return p_.getLong() & 0xffff'ffff'ffff'f018ULL;
157 }
158
159 int64_t hash() const
160 {
161 return idBits();
162 }
163
164 FeatureType type() const
165 {
166 return static_cast<FeatureType>((p_.getUnsignedInt() >> 3) & 3);
167 }
168
169 int typeCode() const
170 {
171 return static_cast<int>((p_.getUnsignedInt() >> 3) & 3);
172 }
173
174 // TODO: not valid for NodeRef
175 int32_t minX() const { return (p_-16).getInt(); }
176 int32_t minY() const { return (p_-12).getInt(); }
177 int32_t maxX() const { return (p_-8).getInt(); }
178 int32_t maxY() const { return (p_-4).getInt(); }
179
180 const char* typeName() const;
181 std::string toString() const;
182
183 bool operator<(const FeaturePtr& other) const noexcept
184 {
185 if(typeCode() != other.typeCode())
186 {
187 return typeCode() < other.typeCode();
188 }
189 return id() < other.id();
190 }
191
192protected:
193 DataPtr p_;
194};
