23public:
24 FeaturePtr() {}
25 FeaturePtr(const uint8_t* p) : p_(p) {}
26 FeaturePtr(const FeaturePtr& other) : p_(other.p_) {}

28 operator DataPtr () const noexcept { return p_; }

30 uint64_t id() const noexcept
31 {
32 uint32_t hi = p_.getUnsignedInt() >> 8;
33 uint32_t lo = (p_ + 4).getUnsignedInt();
34 return (static_cast<uint64_t>(hi) << 32) | lo;
35 }
36
37 uint64_t typedId() const noexcept
38 {
39 return (id() << 2) | typeCode();
40 }
41
42 FeatureHeader header()
43 {
44 return FeatureHeader(p_.getUnsignedLongUnaligned());
45 }
46
47 // TODO: should just get pointer

49 {
50 assert(!isNode());

52 }
53
54 bool isNull() const noexcept { return !p_; }
55 int flags() const noexcept { return p_.getInt(); }
56 bool isArea() const noexcept { return p_.getUnsignedInt() & FeatureFlags::AREA; }
57 bool isNode() const noexcept { return type() == FeatureType::NODE; }
58 bool isWay() const noexcept { return type() == FeatureType::WAY; }
59 bool isRelation() const noexcept { return type() == FeatureType::RELATION; }
60 bool isRelationMember() const noexcept { return flags() & FeatureFlags::RELATION_MEMBER; }
61 bool isType(FeatureTypes types) const noexcept { return types.acceptFlags(flags()); }
62 bool intersects(const Box& bounds) const noexcept
63 {
64 assert(!isNode());
65 return (!((p_-16).getInt() > bounds.maxX() ||
66 (p_-12).getInt() > bounds.maxY() ||
67 (p_-8).getInt() < bounds.minX() ||
68 (p_-4).getInt() < bounds.minY()));
69 }
70
71 TagTablePtr tags() const noexcept
72 {
73 return TagTablePtr::readFrom(p_ + 8);
74 }
75
76 DataPtr ptr() const noexcept { return p_; }
77 DataPtr bodyptr() const noexcept { return (p_ + 12).follow(); }
78
84 DataPtr relationTableFast() const
85 {
86 assert(isRelationMember());
87 // For nodes, the body pointer points to the reltable
88 // For ways and relations, the reltable pointer sits right
89 // before the body anchor
90 DataPtr ppRelTable = isNode() ? (p_ + 12) : (bodyptr() - 4);
91 return ppRelTable.followUnaligned();
92 // TODO: unaligned only for ways/relations
93 }
94
95 int32_t relationTableHandleFast(const uint8_t* base) const
96 {
97 assert(isRelationMember());
98 // For nodes, the body pointer points to the reltable
99 // For ways and relations, the reltable pointer sits right
100 // before the body anchor
101 DataPtr ppRelTable = isNode() ? (p_ + 12) : (bodyptr() - 4);
102 return static_cast<int32_t>((ppRelTable - base) +
103 ppRelTable.getIntUnaligned());
104 }
105
112 int64_t idBits() const
113 {
114 return p_.getLong() & 0xffff'ffff'ffff'ff18;
115 }
116
117 int64_t hash() const
118 {

120 }
121
122 FeatureType type() const
123 {
124 return static_cast<FeatureType>((p_.getUnsignedInt() >> 3) & 3);
125 }
126
127 int typeCode() const
128 {
129 return (p_.getUnsignedInt() >> 3) & 3;
130 }
131
132 // TODO: not valid for NodeRef
133 int32_t minX() const { return (p_-16).getInt(); }
134 int32_t minY() const { return (p_-12).getInt(); }
135 int32_t maxX() const { return (p_-8).getInt(); }
136 int32_t maxY() const { return (p_-4).getInt(); }
137
138 const char* typeName() const;
139 std::string toString() const;
140
141 bool operator<(const FeaturePtr& other) const noexcept
142 {
143 if(typeCode() != other.typeCode())
144 {
145 return typeCode() < other.typeCode();
146 }
147 return id() < other.id();
148 }
149
150protected:
151 DataPtr p_;
152};
