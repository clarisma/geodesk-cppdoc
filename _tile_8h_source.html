174 enum Twin
175 {
176 SELF = 0,
177 NORTH_TWIN = 1,
178 WEST_TWIN = 2,
179 SOUTH_TWIN = 3,
180 EAST_TWIN = 4,
181 INVALID_TWIN = 7
182 };
183
184 uint_fast8_t isTwinOf(Tile other) const
185 {
186 assert(zoom() == other.zoom());
187 uint_fast32_t colDelta = static_cast<uint_fast32_t>(column() - other.column() + 1);
188 uint_fast32_t rowDelta = static_cast<uint_fast32_t>(row() - other.row() + 1);
189 if (colDelta > 2 || rowDelta > 2) return Twin::INVALID_TWIN;
190
191 // Delta is 0 (tile lies west/north), 1 (center) or 2 (tile lies east/south)
192 // We combine the bits into a nibble (colDelta in lower, rowDelta in upper half)
193 // We use the nibble to look up the twin code
194
195 // 98'7654'3210
196 constexpr uint64_t twins = 0x7777'7737'7402'7717;
197 // 0101 (5) = center/self (0)
198 // 0001 (1) = N (1)
199 // 0100 (4) = W (2)
200 // 1001 (9) = S (3)
201 // 0110 (6) = E (4)
202 // all others are invalid (7)
203
204 uint_fast32_t shift = (rowDelta << 4) | (colDelta << 2);
205 // already x4 to address the nibble within twins
206 uint_fast8_t twinCode = static_cast<uint_fast8_t>((twins >> shift) & 15);
207 // assert(twinCode == Twin::INVALID_TWIN || other.twin(twinCode) == *this);
208 return twinCode;
209 }
210
211 Tile twin(uint_fast8_t twinCode) const
212 {
213 assert(twinCode >= 0 && twinCode <= 4);
214
215 // 4 3 2 1 0
216 constexpr uint32_t offsets = 0b0110'1001'0100'0001'0101;
217 uint_fast32_t nibble = (offsets >> (twinCode * 4)) & 15;
218 int colDelta = static_cast<int>(nibble & 3) - 1;
219 int rowDelta = static_cast<int>(nibble >> 2) - 1;
220 Tile twin = fromColumnRowZoom(
221 column() + colDelta, row() + rowDelta, zoom());
222 assert(twin.isTwinOf(*this) == twinCode);
223 return twin;
224 }
225
226protected:
227 static constexpr uint32_t EMPTY = 0xffff'ffff;
228
229 constexpr Tile(int col, int row, ZoomLevel zoom)
230 : tile_((zoom << 24) | (row << 12) | col) {}
231
232 explicit Tile(int t) : tile_(t) {}
233
234 uint32_t tile_;
235};
