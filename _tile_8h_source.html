128 Tile neighbor(int colDelta, int rowDelta) const
129 {
130 int z = zoom();
131 int x = column();
132 int y = row();
133 int mask = (1 << z) - 1;
134 x = (x + colDelta) & mask;
135 y = (y + rowDelta) & mask;
136 return fromColumnRowZoom(x, y, z);
137 }

168 {
169 const int currentZoom = this->zoom();
170 assert(lowerZoom <= currentZoom); // Can't zoom out to higher level

175 enum Twin
176 {
177 SELF = 0,
178 NORTH_TWIN = 1,
179 WEST_TWIN = 2,
180 SOUTH_TWIN = 3,
181 EAST_TWIN = 4,
182 INVALID_TWIN = 7
183 };
184
185 uint_fast8_t isTwinOf(Tile other) const
186 {
187 assert(zoom() == other.zoom());
188 uint_fast32_t colDelta = static_cast<uint_fast32_t>(column() - other.column() + 1);
189 uint_fast32_t rowDelta = static_cast<uint_fast32_t>(row() - other.row() + 1);
190 if (colDelta > 2 || rowDelta > 2) return Twin::INVALID_TWIN;
191
192 // Delta is 0 (tile lies west/north), 1 (center) or 2 (tile lies east/south)
193 // We combine the bits into a nibble (colDelta in lower, rowDelta in upper half)
194 // We use the nibble to look up the twin code
195
196 // 98'7654'3210
197 constexpr uint64_t twins = 0x7777'7737'7402'7717;
198 // 0101 (5) = center/self (0)
199 // 0001 (1) = N (1)
200 // 0100 (4) = W (2)
201 // 1001 (9) = S (3)
202 // 0110 (6) = E (4)
203 // all others are invalid (7)
204
205 uint_fast32_t shift = (rowDelta << 4) | (colDelta << 2);
206 // already x4 to address the nibble within twins
207 uint_fast8_t twinCode = static_cast<uint_fast8_t>((twins >> shift) & 15);
208 // assert(twinCode == Twin::INVALID_TWIN || other.twin(twinCode) == *this);
209 return twinCode;
210 }
211
212 Tile twin(uint_fast8_t twinCode) const
213 {
214 assert(twinCode >= 0 && twinCode <= 4);
215
216 // 4 3 2 1 0
217 constexpr uint32_t offsets = 0b0110'1001'0100'0001'0101;
218 uint_fast32_t nibble = (offsets >> (twinCode * 4)) & 15;
219 int colDelta = static_cast<int>(nibble & 3) - 1;
220 int rowDelta = static_cast<int>(nibble >> 2) - 1;
221 Tile twin = fromColumnRowZoom(
222 column() + colDelta, row() + rowDelta, zoom());
223 assert(twin.isTwinOf(*this) == twinCode);
224 return twin;
225 }
226
227protected:
228 static constexpr uint32_t EMPTY = 0xffff'ffff;
229
230 constexpr Tile(int col, int row, ZoomLevel zoom)
231 : tile_((zoom << 24) | (row << 12) | col) {}
232
233 explicit Tile(int t) : tile_(t) {}
234
235 uint32_t tile_;
236};
