181 TilePair zoomedOut(int lowerZoom) const
182 {
183 assert(lowerZoom <= this->zoom()); // Can't zoom out to higher level
184 uint32_t northWest = static_cast<uint32_t>(first().zoomedOut(lowerZoom));
185 uint32_t southEast = static_cast<uint32_t>(second().zoomedOut(lowerZoom));
186 return TilePair(northWest | ((northWest == southEast) ? 0 :
187 tile_ & (EXTENDS_EAST | EXTENDS_SOUTH)));
188 }

212 uint_fast8_t isTwinOf(Tile other)
213 {
214 assert(zoom() == other.zoom());
215 uint_fast32_t colDelta = static_cast<uint_fast32_t>(other.column() - column());
216 uint_fast32_t rowDelta = static_cast<uint_fast32_t>(other.row() - row());
217 if (colDelta > 1 || rowDelta > 1) return Twin::INVALID_TWIN;
218
219 /*

226 uint_fast32_t shift =
227 ((tile_ >> (EXTENDS_EAST_BIT - 4)) & 0x30) |
228 (rowDelta << 3) | (colDelta << 2);
229 //already x4 to address nibble within twins
230
231 // A98'7654'3210
232 constexpr uint64_t twins = 0x7777'7173'7724'7770;
233 // 0000 (0) = self/no twin (0)
234 // 1010 (10) = north twin (1)
235 // 0101 (5) = west twin (2)
236 // 1000 (8) = south twin (3)
237 // 0100 (6) = east twin (4)
238 // all others are invalid (7)
239
240 uint_fast8_t twinCode = static_cast<uint_fast8_t>((twins >> shift) & 15);
241 assert(twinCode == Twin::INVALID_TWIN || TilePair::covering(other, twinCode) == *this);
242 return twinCode;
243 }

245 static TilePair covering(Tile tile, uint_fast8_t twinCode)
246 {
247 assert(twinCode >= 0 && twinCode <= 4);
248
249 // 4 3 2 1 0
250 constexpr uint32_t offsets = 0b0100'1000'0101'1010'0000;
251 uint_fast32_t nibble = offsets >> (twinCode * 4);
252 int colDelta = static_cast<int>(nibble & 1);
253 int rowDelta = static_cast<int>((nibble >> 1) & 1);
254 uint_fast32_t southEastBits = (nibble & 12) << (EXTENDS_EAST_BIT - 2);
255 TilePair pair = TilePair(static_cast<uint32_t>(fromColumnRowZoom(
256 tile.column() - colDelta, tile.row() - rowDelta, tile.zoom())) |
257 southEastBits);
258 // assert(pair.isTwinOf(tile) == twinCode);
259 return pair;
