99 bool contains(int32_t x, int32_t y) const
100 {
101 if (minX() > maxX())
102 {
103 // empty or Antimeridian-crossing
104 if (minY() > maxY()) return false; // empty box cannot contain anything
105 // If 180 longitude crossed, minX and maxY are swapped
106 return (x >= maxX() && x <= minX() &&
107 y >= minY() && y <= maxY());
108 }
109 return containsSimple(x, y);
110 }

112 bool containsSimple(int32_t x, int32_t y) const

115 }

138 bool contains(const Box& other) const
139 {
140 bool isSimple = minX() <= maxX();
141 bool isOtherSimple = other.minX() <= other.maxX();
142 if (isSimple && isOtherSimple) return containsSimple(other);
143
144 // Either of the boxes is empty --> cannot contain or be contained
145 if (minY() < maxY() || other.minY() < other.maxY()) return false;
146
147 return std::min(other.minX(), other.maxX()) >= std::min(minX(), maxX()) &&
148 std::max(other.minX(), other.maxX()) <= std::max(minX(), maxX()) &&
149 other.minY() >= minY() && other.maxY() <= maxY();
150 }

158 inline static Box simpleIntersection(const Box& a, const Box& b)
159 {
160 assert(a.isSimple());
161 assert(b.isSimple());
162
163 int x1 = std::max(a.minX(), b.minX());
164 int y1 = std::max(a.minY(), b.minY());
165 int x2 = std::min(a.maxX(), b.maxX());
166 int y2 = std::min(a.maxY(), b.maxY());
167
168 if (x2 < x1 || y2 < y1) return Box(); // no intersection
169 return Box(x1, y1, x2, y2);
170 }

319 static int trimmedSubtract(int32_t x, int32_t y)
320 {
321 int r = x - y;
322 if (((x ^ y) & (x ^ r)) < 0) return INT_MIN;
323 return r;
324 }
325
333 static inline int trimmedAdd(int32_t x, int32_t y)
334 {
335 int r = x + y;
336 if (((x ^ r) & (y ^ r)) < 0) return INT_MAX;
337 return r;
338 }
