311 template<typename Stream>
312 void format(Stream& out) const
313 {
314 switch (type())
315 {
316 case 1: // global string
317 case 3: // local string (fall through)
318 stringValue_.format(out);
319 break;
320 case 0: // narrow number
321 {
322 // TODO: make more efficient by encoding in reverse
323 char buf[32];
324 char* end = clarisma::Format::integer(buf,
325 TagValues::intFromNarrowNumber(rawNumberValue()));
326 out.write(buf, end - buf);
327 break;
328 }
329 case 2: // wide number
330 TagValues::decimalFromWideNumber(
331 rawNumberValue()).format(out);
332 break;
333 default:
334 UNREACHABLE_CASE
335 }
336 }
337
338private:
339 int type() const { return static_cast<int>(taggedNumberValue_) & 3; }
340 uint_fast32_t rawNumberValue() const
341 {
342 return static_cast<uint_fast32_t>(taggedNumberValue_ >> 2);
343 }
344
345 uint64_t taggedNumberValue_;
346 StringValue stringValue_;
347};
