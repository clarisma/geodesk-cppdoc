27 template<typename LT>
28 StringSet(const StringList<LT>& list)
29 {
30 // first, we calculate the total size taken up by the entries
31
32 size_t count = list.size();
33 size_t stringAreaSize = 0;
34 for(int i=0; i<count; i++)
35 {
36 stringAreaSize += entrySize(list.get(i).length());
37 }
38
39 // Now we select a reasonable table size (twice the number
40 // of items, rounded up to the nearest power-of-2)
41
42 tableSize_ = Bytes::roundUpToPowerOf2(count * 2);
43 uint32_t tableSizeInBytes = tableSize_ * sizeof(uint32_t);
44 data_ = new uint8_t[tableSizeInBytes + stringAreaSize];
45 clearTable();
46
47 // Finally, we create the entries and put them into the
48 // hashtable
49
50 uint32_t* pb = data_.get() + tableSizeInBytes;
51 for(int i=0; i<count; i++)
52 {
53 Entry* pe = reinterpret_cast<Entry*>(pb);
54 pe->next = 0;
55 std::string_view s = list.get(i);
56 pe->string.init(s.data(), s.length());
57 insert(pe);
58 pb += entrySize(p->length());
59 p++;
60 }
61 }
