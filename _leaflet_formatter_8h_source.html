57 void writeCollectionRelationGeometry(clarisma::Buffer& out, FeatureStore* store, RelationPtr rel) const
58 {
59 out.write("L.featureGroup([");
60 uint64_t count = 0;
61 bool isFirst = true;
62 FastMemberIterator iter(store, rel);
63 for (;;)
64 {
65 FeaturePtr member = iter.next();
66 if (member.isNull()) break;
67 if (!isFirst) out.writeByte(',');
68 isFirst = false;
69 int memberType = member.typeCode();
70 if (memberType == 1)
71 {
72 self().writeWayGeometry(out, WayPtr(member));
73 }
74 else if (memberType == 0)
75 {
76 self().writeNodeGeometry(out, NodePtr(member));
77 }
78 else
79 {
80 assert(memberType == 2);
81 self().writeRelationGeometry(out, store, RelationPtr(member));
82 }
83 out.writeByte(')');
84 }
85 out.writeByte(']');
