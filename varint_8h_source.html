13inline uint64_t readVarint35(const uint8_t*& p)
14{
15 uint32_t val;
16 uint8_t b;
17 b = *p++;
18 val = b & 0x7f;
19 if ((b & 0x80) == 0) return val;
20 b = *p++;
21 val |= static_cast<uint32_t>(b & 0x7f) << 7;
22 if ((b & 0x80) == 0) return val;
23 b = *p++;
24 val |= static_cast<uint32_t>(b & 0x7f) << 14;
25 if ((b & 0x80) == 0) return val;
26 b = *p++;
27 val |= static_cast<uint32_t>(b & 0x7f) << 21;
28 if ((b & 0x80) == 0) return val;
29 b = *p++;
30 val |= static_cast<uint32_t>(b & 0x7f) << 28;
31 assert((b & 0x80) == 0);
32 return val;
33}

82inline int32_t readSignedVarint32(const uint8_t*& p)
83{
84 int64_t val = static_cast<int64_t>(readVarint35(p));

103inline int countVarints(const void* pStart, const void* pEnd)
104{
105 int count = 0;
106 const char* p = reinterpret_cast<const char*>(pStart);
107 while (p < pEnd)
108 {
109 if (*p++ >= 0) count++;
110 }
111 return count;

115{

144inline void writeVarint(uint8_t*& p, uint64_t val)
145{
146 while (val >= 0x80)
147 {
148 *p++ = (val & 0x7f) | 0x80;
149 val >>= 7;
150 }
151 *p++ = static_cast<uint8_t>(val);
152}
