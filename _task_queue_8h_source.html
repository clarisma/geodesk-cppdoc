16template <typename Context, typename Task>
17class TaskQueue
18{
19public:
20 explicit TaskQueue(int size) :
21 size_(size),
22 count_(0),
23 front_(0),
24 rear_(0),
25 running_(true)
26 {
27 assert(size > 0);
28 queue_.resize(size);
29 }
30
31 // TODO: rename to "submit()"
32 void post(Task&& task)
33 {
34 std::unique_lock<std::mutex> lock(mutex_);
35 if (count_ == size_)
36 {
37 /*
38 printf("Thread %s: Queue %p full, waiting for space...\n",
39 Threads::currentThreadId().c_str(), this);
40 */
41 notFull_.wait(lock, [this] { return count_ < size_; });
42 }
43 queue_[rear_] = std::move(task);
44 rear_ = (rear_ + 1) % size_;
45 count_++;
46 notEmpty_.notify_one();
47 }
48
49 // TODO: rename to "trySubmit()"
50 bool tryPost(Task&& task)
51 {
52 std::unique_lock<std::mutex> lock(mutex_);
53 if (count_ == size_) return false;
54 queue_[rear_] = std::move(task);
55 rear_ = (rear_ + 1) % size_;
56 count_++;
57 notEmpty_.notify_one();
58 return true;
59 }
60
61 // TODO: Needs test
62 bool fill(std::function<bool(Task*)> supplier)
63 {
64 std::unique_lock<std::mutex> lock(mutex_);
65
66 int tasksAdded = 0;
67 while (count_ < size_)
68 {
69 if (!supplier(&queue_[rear_])) break;
70 // Successfully got a new task directly into the queue
71 rear_ = (rear_ + 1) % size_;
72 count_++;
73 tasksAdded++;
74 }
75
76 if (tasksAdded) notEmpty_.notify_all();
77 return count_ == size_; // Return true if the queue is full, indicating there might be more tasks to add
78 }
79
80
81 int minimumRemainingCapacity()
82 {
83 std::unique_lock<std::mutex> lock(mutex_);
84 // TODO: No lock needed, as long as there aren't multiple consumers
85 // count_ will only decrease, so this should be safe without locking
86 // But no noticeable performance difference, so we'll leave the lock for now
87 return size_ - count_;
88 }
89
90 void process(Context* ctx)
91 {
92 for(;;)
93 {
94 Task task;
95 {
96 std::unique_lock<std::mutex> lock(mutex_);
97 for (;;)
98 {
99 if (!running_)
100 {
101 // Console::debug("Finished processing queue %p.", this);
102 return;
103 }
104 if (count_ > 0) break;
105 /*
106 printf("Thread %s: Waiting for tasks in queue %p...\n",
107 Threads::currentThreadId().c_str(), this);
108 */
109 notEmpty_.wait(lock);
110 }
111 task = std::move(queue_[front_]);
112 front_ = (front_ + 1) % size_;
113 count_--;
114 notFull_.notify_one();
115 }
116 ctx->processTask(task);
117 }
118 }
119
120
121 void awaitCompletion()
122 {
123 //LOG("Awaiting completion of queue %p...", this);
124 std::unique_lock<std::mutex> lock(mutex_);
125 while (count_ != 0)
126 { // Continue to wait as long as there are tasks in the queue
127 //LOG("There are still %d task(s) in queue %p...", count_, this);
128 notFull_.wait(lock); // Wait for a signal that a task has been completed
129 }
130 //LOG("Queue %p is empty.", this);
131 }
132
133 void shutdown()
134 {
135 //LOG("Shutting down queue %p...", this);
136 std::unique_lock<std::mutex> lock(mutex_);
137 running_ = false;
138 notEmpty_.notify_all();
139 }
140
141private:
142 std::vector<Task> queue_;
143 std::mutex mutex_;
144 std::condition_variable notEmpty_, notFull_;
145 int front_;
146 int rear_;
147 int size_;
148 int count_;
149 bool running_;
150};
